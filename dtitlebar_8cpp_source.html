253 void DTitlebarPrivate::updateFullscreen()
254 {
255  D_Q(DTitlebar);
256 
257  if (!autoHideOnFullscreen) {
258  return;
259  }
260 
261  bool isFullscreen = targetWindow()->windowState().testFlag(Qt::WindowFullScreen);
262  auto mainWindow = qobject_cast<DMainWindow *>(targetWindow());
263  if (!isFullscreen) {
264  quitFullButton->hide();
265  mainWindow->setMenuWidget(q);
266  showOnFullscreen();
267  } else {
268  // must set to empty
269  quitFullButton->show();
270  if (mainWindow->menuWidget()) {
271  mainWindow->menuWidget()->setParent(nullptr);
272  mainWindow->setMenuWidget(Q_NULLPTR);
273  }
274 
275  q->setParent(mainWindow);
276  q->show();
277  hideOnFullscreen();
278  }

281 void DTitlebarPrivate::updateButtonsState(Qt::WindowFlags type)
282 {
283  D_Q(DTitlebar);
284  bool useDXcb = DPlatformWindowHandle::isEnabledDXcb(targetWindow());
285  bool isFullscreen = targetWindow()->windowState().testFlag(Qt::WindowFullScreen);
286 
287  bool showTitle = (type.testFlag(Qt::WindowTitleHint) || !useDXcb) && !embedMode;
288  if (titleLabel) {
289  titleLabel->setVisible(showTitle);
290  }
291  if (iconLabel) {
292  iconLabel->setVisible(showTitle);
293  }
294 
295  // Never show in embed/fullscreen
296  bool forceHide = embedMode || isFullscreen;
297 
298  bool showMin = (type.testFlag(Qt::WindowMinimizeButtonHint) || !useDXcb) && !forceHide;
299  minButton->setVisible(showMin);
300 
301  bool allowResize = true;
302  if (q->window() && q->window()->windowHandle()) {
303  auto functions_hints = DWindowManagerHelper::getMotifFunctions(q->window()->windowHandle());
304  allowResize = functions_hints.testFlag(DWindowManagerHelper::FUNC_RESIZE);
305  }
306 
307  bool showMax = (type.testFlag(Qt::WindowMaximizeButtonHint) || !useDXcb) && !forceHide && allowResize;
308  maxButton->setVisible(showMax);
309 
310  bool showClose = (type.testFlag(Qt::WindowCloseButtonHint) || !useDXcb) && !forceHide;
311  closeButton->setVisible(showClose);
312 
313  bool showOption = (type.testFlag(Qt::WindowSystemMenuHint) || !useDXcb) && !isFullscreen;
314  optionButton->setVisible(showOption);
315 
316  buttonArea->adjustSize();
317  buttonArea->resize(buttonArea->size());
318 
319  if (titlePadding) {
320  titlePadding->setFixedSize(buttonArea->size());
321  }
322 }

640 bool DTitlebar::eventFilter(QObject *obj, QEvent *event)
641 {
642  D_D(DTitlebar);
643 
644  if (obj == d->targetWindow()) {
645  switch (event->type()) {
646  case QEvent::ShowToParent:
647  d->handleParentWindowIdChange();
648  d->updateButtonsFunc();
649  d->updateButtonsState(d->targetWindow()->windowFlags());
650  break;
651  case QEvent::Resize:
652  if (d->autoHideOnFullscreen) {
653  setFixedWidth(d->targetWindow()->width());
654  }
655  break;
656  case QEvent::HoverMove: {
657  auto mouseEvent = reinterpret_cast<QMouseEvent *>(event);
658  bool isFullscreen = d->targetWindow()->windowState().testFlag(Qt::WindowFullScreen);
659  if (isFullscreen && d->autoHideOnFullscreen) {
660  if (mouseEvent->pos().y() > height() && d->isVisableOnFullscreen()) {
661  d->hideOnFullscreen();
662  }
663  if (mouseEvent->pos().y() < 2) {
664  d->showOnFullscreen();
665  }
666  }
667  break;
668  }
669  case QEvent::WindowStateChange: {
670  d->handleParentWindowStateChange();
671  break;
672  }
673  default:
674  break;
675  }
676  }
677  return QWidget::eventFilter(obj, event);
678 }

718 void DTitlebar::setCustomWidget(QWidget *w, Qt::AlignmentFlag wflag, bool fixCenterPos)
719 {
720  D_D(DTitlebar);
721  if (!w || w == d->titleArea) {
722  return;
723  }

726 
727  QHBoxLayout *l = new QHBoxLayout;
728  l->setSpacing(0);
729  l->setMargin(0);
730 
731  if (fixCenterPos) {
732  d->titlePadding = new QWidget;
733  d->titlePadding->setFixedSize(old);
734  l->addWidget(d->titlePadding);
735  }
736 
737  l->addWidget(w, 0, wflag);
738  qDeleteAll(d->coustomAtea->children());
739  d->titleLabel = Q_NULLPTR;
740  d->titleArea = Q_NULLPTR;
741  d->iconLabel = Q_NULLPTR;
742  d->titlePadding = Q_NULLPTR;
743  d->coustomAtea->setLayout(l);
744  d->buttonArea->resize(old);
745  d->customWidget = w;
746 
747  w->resize(d->coustomAtea->size());
748 }
